### 使用

```tsx
import Popup from '@/components/Popup';
import { useState } from 'react';

const [visible, setVisible] = useState(false);

<Popup
  destroyOnClosed
  visible={visible}
  round
  position='bottom'
  safeAreaInsetBottom
  onPressOverlay={() => {
    setVisible(false);
  }}
>
  <Popup.Header title='每次打开都是新的子元素' />
  <Popup.KeyboardShim />
</Popup>;
```
