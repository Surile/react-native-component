### 使用

```tsx

const showStepSelector = () => {
  StepSelector({
    request: request,
    onConfirm: (v, o, isEnd) => {
      console.log('v => ', v);
      console.log('o => ', o);
      console.log('isEnd => ', isEnd);
    },
  }).catch(() => {});
};
```
