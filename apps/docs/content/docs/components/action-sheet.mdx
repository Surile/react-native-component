### 使用

```tsx
import Button from "@/components/button";
import ActionSheet from "@/components/action-sheet";
import { Text } from "react-native";

<Button
  onPress={() => {
    ActionSheet({
      actions: ["选项一", "选项二"],
    })
      .then(({ item, index }) => {
        console.log("提示弹窗 -> 选项");
        console.log(item);
        console.log(index);
      })
      .catch((e) => {
        console.log("提示弹窗 -> 关闭");
        console.log(e);
      });
  }}
>
  <Text>按钮</Text>
</Button>;
```
