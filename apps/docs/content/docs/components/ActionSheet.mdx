### 使用

```tsx
import ActionSheet from '@/components/ActionSheet';

const show = () => {
  ActionSheet({
    actions: ['选项一', '选项二'],
  })
    .then(({ item, index }) => {
      console.log('提示弹窗 -> 选项');
      console.log(item);
      console.log(index);
    })
    .catch((e) => {
      console.log('提示弹窗 -> 关闭');
      console.log(e);
    });
};
```
